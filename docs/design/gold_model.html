Gold Layer Dimensional Model

Enterprise Data Warehouse Star Schemas

1. Sales & Order Fulfillment (fact_sales)

Tracks revenue, discounts, and product sales at the order line level.

@startuml
skinparam linetype ortho
skinparam backgroundColor transparent
skinparam entityBorderColor #DAA520
skinparam entityBackgroundColor #FEFECE
hide circle

entity dim_date {
    *dim_date_key : INT
    --
    full_date : DATE
    day_of_week : INT
    day_name : VARCHAR
    day_of_month : INT
    day_of_year : INT
    week_of_year : INT
    month_number : INT
    month_name : VARCHAR
    quarter : INT
    year : INT
    is_weekend : BOOLEAN
    is_holiday : BOOLEAN
}

entity dim_customer {
    *dim_customer_key : BIGINT
    --
    *customer_id : BIGINT
    email : STRING
    phone : STRING
    full_name : STRING
    gender : STRING
    customer_type : STRING
    status : STRING
    marketing_opt_in : BOOLEAN
    registration_date : TIMESTAMP
    _valid_from : TIMESTAMP
    _valid_to : TIMESTAMP
    _is_current : BOOLEAN
    _version : INT
}

entity dim_product {
    *dim_product_key : BIGINT
    --
    *inventory_item_id : BIGINT
    sku : STRING
    product_name : STRING
    category_name : STRING
    brand_name : STRING
    unit_cost : DECIMAL
}

entity dim_location {
    *dim_location_key : BIGINT
    --
    *address_id : BIGINT
    city : STRING
    state : STRING
    postal_code : STRING
    country : STRING
    city_tier : STRING
}

entity dim_campaign {
    *dim_campaign_key : BIGINT
    --
    *campaign_id : INT
    campaign_name : STRING
    channel : STRING
    sub_channel : STRING
    status : STRING
    start_date : DATE
    end_date : DATE
}

entity fact_sales {
    *order_id : BIGINT
    *line_id : BIGINT
    --
    *dim_date_key : INT
    *dim_customer_key : BIGINT
    *dim_product_key : BIGINT
    *dim_location_key : BIGINT
    *dim_campaign_key : BIGINT
    order_number : STRING
    order_status : STRING
    quantity : INT
    line_total : DECIMAL
    allocated_discount_amount : DECIMAL
    allocated_tax_amount : DECIMAL
    unit_price : DECIMAL
}

dim_date ||--o{ fact_sales : "has"
dim_customer ||--o{ fact_sales : "has"
dim_product ||--o{ fact_sales : "has"
dim_location ||--o{ fact_sales : "has"
dim_campaign ||--o{ fact_sales : "has"
@enduml

Fact Table Details: fact_sales

Column Name Data Type Type Description
dim_date_keyINTFKForeign key to dim_date
dim_customer_keyBIGINTFKForeign key to dim_customer
dim_product_keyBIGINTFKForeign key to dim_product
dim_location_keyBIGINTFKForeign key to dim_location
dim_campaign_keyBIGINTFKForeign key to dim_campaign
order_idBIGINTDegenerate DimSource system order ID
line_idBIGINTDegenerate DimSource system line ID
quantityINTAdditive MeasureNumber of units purchased
line_totalDECIMAL(15,2)Additive MeasureTotal amount for the line
allocated_discount_amountDECIMAL(15,2)Additive MeasureHeader discount allocated to line
unit_priceDECIMAL(15,2)Non-AdditivePrice per unit
